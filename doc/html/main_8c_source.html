164 int ParseMemArgument(MemImage *m) {
165  int i;
166  char *addr = NULL;
167  char *len = NULL;
168  char *endPtr;
169 
170  // Set address and length to default values
171  m->len = m->addr = 0;
172 
173  // First argument is name. Find where it ends
174  for (i = 0; i < (MAX_FILELEN + 1) && m->file[i] != '\0' &&
175  m->file[i] != ':'; i++);
176  // Check if end reached without finding end of string
177  if (i == (MAX_FILELEN + 1)) return 1;
178  if (m->file[i] == '\0') return 0;
179 
180  // End of token marker, search address
181  m->file[i++] = '\0';
182  addr = m->file + i;
183  for (; i < (MAX_FILELEN + 1) && m->file[i] != '\0' && m->file[i] != ':';
184  i++);
185  // Check if end reached without finding end of string
186  if (i == MAX_FILELEN + 1) return 1;
187  // If end of token marker, search length
188  if (m->file[i] == ':') {
189  m->file[i++] = '\0';
190  len = m->file + i;
191  // Verify there's an end of string
192  for (; i < (MAX_FILELEN + 1) && m->file[i] != '\0'; i++);
193  if (m->file[i] != '\0') return 1;
194  }
195  // Convert strings to numbers and return
196  if (addr && *addr) m->addr = strtol(addr, &endPtr, 0);
197  if (m->addr == 0 && addr == endPtr) return 2;
198  if (len && *len) m->len = strtol(len, &endPtr, 0);
199  if (m->len == 0 && len == endPtr) return 3;
200 
201  return 0;
202 }

215 int ParseMemRange(char inStr[], uint32_t *addr, uint32_t *len) {
216  int32_t i;
217  char *saddr, *endPtr;
218  char scratch;
219  long val;
220 
221  // Seek end of string or field separator (:)
222  for (i = 0; (i < (MAX_MEM_RANGE + 1)) && (inStr[i] != '\0') &&
223  (inStr[i] != ':'); i++);
224 
225  if (i == (MAX_MEM_RANGE + 1)) return 1;
226  // Store end of string or separator, and ensure proper end of string
227  scratch = inStr[i];
228  inStr[i++] = '\0';
229  // Convert to long
230  val = strtol(inStr, &endPtr, 0);
231  if (*endPtr != '\0' || val < 0) return 1;
232  *addr = val;
233  // If we had field separator, repeat scan for length
234  if (scratch == '\0') return 0;
235  saddr = inStr + i;
236  for (; (i < (MAX_MEM_RANGE + 1)) && (inStr[i] != '\0'); i++);
237  if (i == (MAX_MEM_RANGE + 1)) return 1;
238  val = strtol(saddr, &endPtr, 0);
239  if (*endPtr != '\0' || val < 0) return 1;
240  *len = val;
241  return 0;
242 }

290 uint16_t *AllocAndFlash(MemImage *fWr, int autoErase, int noPatch,
291  int columns) {
292  FILE *rom;
293  uint16_t *writeBuf;
294  uint32_t addr;
295  int toWrite;
296  uint32_t i;
297  // Address string, e.g.: 0x123456
298  char addrStr[9];
299 
300  // If header covered by range, but not completeley, reject flash command
301  if (!autoErase && ((fWr->addr < ROM_HEAD_LEN && fWr->addr) ||
302  (!fWr->addr && fWr->len < ROM_HEAD_LEN))) return NULL;
303  // Open the file to flash
304  if (!(rom = fopen(fWr->file, "rb"))) {
305  perror(fWr->file);
306  return NULL;
307  }
308 
309  // Obtain length if not specified
310  if (!fWr->len) {
311  fseek(rom, 0, SEEK_END);
312  fWr->len = ftell(rom);
313  fseek(rom, 0, SEEK_SET);
314  }
315 
316  writeBuf = malloc(fWr->len);
317  if (!writeBuf) {
318  perror("Allocating write buffer RAM");
319  fclose(rom);
320  return NULL;
321  }
322  fread(writeBuf, fWr->len, 1, rom);
323  fclose(rom);
324  // If requested, perform auto-erase
325  if (autoErase) {
326  printf("Auto-erasing range 0x%06X:%06X...\n", fWr->addr, fWr->len);
327  if (WfFlashErase(fWr->addr, fWr->len)) {
328  free(writeBuf);
329  PrintErr("Auto-erase failed!\n");
330  return NULL;
331  }
332  }
333 
334  // If header is included in flash image, and unless prohibited
335  if (!fWr->addr && !noPatch) RomHeadPatch((uint8_t*)writeBuf);
336 
337  printf("Flashing ROM %s starting at 0x%06X...\n", fWr->file, fWr->addr);
338 
339  for (i = 0, addr = fWr->addr; i < fWr->len;) {
340 // toWrite = MIN(2*1152, fWr->len - i);
341 // toWrite = MIN(57600, fWr->len - i);
342 // toWrite = MIN(1440, fWr->len - i);
343  toWrite = MIN(64800, fWr->len - i);
344  if (WfFlash(addr, toWrite, ((uint8_t*)writeBuf) + i)) {
345  free(writeBuf);
346  PrintErr("Couldn't write to cart!\n");
347  return NULL;
348  }
349  // Update vars and draw progress bar
350  i += toWrite;
351  addr += toWrite;
352  sprintf(addrStr, "0x%06X", addr);
353  ProgBarDraw(i, fWr->len, columns, addrStr);
354  }
355  putchar('\n');
356  return writeBuf;
357 }
