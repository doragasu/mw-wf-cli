47 int WfConnect(char host[], uint16_t port) {
48  const struct addrinfo hints = {
49  .ai_family = AF_INET,
50  .ai_socktype = SOCK_STREAM,
51  };
52  struct addrinfo *srvInfo;
53  char strPort[6];
54  int flag = 1;
55 
56  // DNS lookup code
57  snprintf(strPort, 5, "%d", port);
58  strPort[5] = '\0';
59  if ((getaddrinfo(host, strPort, &hints, &srvInfo) != 0) || (!srvInfo)) {
60  PrintErr("DNS error for %s:%s\n", host, strPort);
61  if (srvInfo) freeaddrinfo(srvInfo);
62  return WF_ERROR;
63  }
64  d.srvAddr = &((struct sockaddr_in *)srvInfo->ai_addr)->sin_addr;
65 
66  // Create socket ...
67  d.sock = socket(srvInfo->ai_family, srvInfo->ai_socktype, 0);
68  if (d.sock < 0) {
69  freeaddrinfo(srvInfo);
70  PrintErr("Could not create socket!\n");
71  return WF_ERROR;
72  }
73  // Disable Nagle algorithm
74  if (setsockopt(d.sock, IPPROTO_TCP, TCP_NODELAY, (char*)&flag,
75  sizeof(int))) {
76  freeaddrinfo(srvInfo);
77  close(d.sock);
78  PrintErr("Could not set socket options!\n");
79  }
80  // ... and connect!
81  if (connect(d.sock, srvInfo->ai_addr, srvInfo->ai_addrlen) != 0) {
82  close(d.sock);
83  freeaddrinfo(srvInfo);
84  PrintErr("Could not connect to %s:%s.\n", host, strPort);
85  return WF_ERROR;
86  }
87 
88  d.connected = TRUE;
89  // Connection succesful!
90  freeaddrinfo(srvInfo);
91  return WF_OK;

245 uint32_t WfRead(uint32_t addr, uint32_t len, uint8_t buf[]) {
246  uint32_t total;

249  // Reading uses two stages:

261  return 0;
262  }
263  // Receive data
264  total = 0;
265  while ((recvd = recv(d.sock, buf + total, len - total, 0)) > 0) {
266  total += recvd;
267  }
